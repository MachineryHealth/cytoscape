30  package org.cytoscape.equations;
31  
32  
33  import java.util.List;
34  
35  
36  public interface Function {
37  	/***
38  	 *  Used to parse the function string.  This name is treated in a case-insensitive manner!
39  	 *  @return the name by which you must call the function when used in an attribute equation.
40  	 */
41  	String getName();
42  
43  	/***
44  	 *  Used to provide help for users.  Unlike getUsageDescription(), this is an informal English description,
45  	 *  like "Calculates the sine of its argument."
46  	 *
47  	 *  @return a description of what this function does
48  	 */
49  	String getFunctionSummary();
50  
51  	/***
52  	 *  Used to provide help for users.  Unlike getFunctionSummary(), this describes how to call this function,
53  	 *  like "Call with SIN(number)."
54  	 *
55  	 *  @return a description of how to use this function
56  	 */
57  	String getUsageDescription();
58  
59  	/***
60  	 *  @return the static return type of this function, Object.class, Double.cLass, String.class, or Boolean.class.
61  	 *           If the static return type is Object.class, the dynamic return type will be one of Double.cLass, String.class, or Boolean.class
62  	 *           and will depend on the arguments passed to the function!
63  	 *
64  	 *  Note, this is used by external tools used to filter a list of functions based on what a valid return type might be.
65  	 *  In Cytoscape it is used in the attribute browser's formula builder.
66  	 */
67  	Class getReturnType();
68  
69  	/***
70  	 *  @return the return type for this function (Double.cLass, String.class, or Boolean.class)
71  	 *           or null if the args passed in had the wrong arity or a type mismatch
72  	 *
73  	 *  Note that this is different from getReturnType() in that it will never return the wildcard Object.class.
74  	 *  It is used by the parser which knows the actual type of the arguments in any given call to this function.
75  	 */
76  	Class validateArgTypes(final Class[] argTypes);
77  
78  	/***
79  	 *  Used to invoke this function.
80  	 *  @param args the function arguments which must correspond in type and number to what getParameterTypes() returns.
81  	 *  @return the result of the function evaluation.  The actual type of the returned object will be what getReturnType() returns.
82  	 *  @throws ArithmeticException thrown if a numeric error, e.g. a division by zero occurred.
83  	 *  @throws IllegalArgumentException thrown for any error that is not a numeric error, for example if a function only accepts positive numbers and a negative number was passed in.
84  	 */
85  	Object evaluateFunction(final Object[] args) throws FunctionError;
86  
87  	/***
88  	 *  Used with the equation builder.
89  	 *
90  	 *  @param leadingArgs the types of the arguments that have already been selected by the user.
91  	 *  @return the set of arguments (must be a collection of String.class, Long.class, Double.class, Boolean.class and List.class) that are candidates for the next argument.  A null return indicates that no further arguments are valid.
92  	 *  Please note that if the returned set contains a null, this indicates an optional additional argument.
93  	 */
94  	List<Class> getPossibleArgTypes(final Class[] leadingArgs);
95  }
