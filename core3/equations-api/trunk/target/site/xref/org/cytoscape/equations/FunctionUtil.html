52  	/***
53  	 *  Assumes that "arg" is a "String", "Boolean", "Long" or a "Double and converts it to "double".
54  	 *  @return the converted argument as a "double"
55  	 *  @throws IllegalArgumentException if the argument cannot be converted to a "double"
56  	 */
57  	static public double getArgAsDouble(final Object arg) throws IllegalArgumentException {
58  		if (arg.getClass() == Double.class)
59  			return (Double)arg;
60  		if (arg.getClass() == Long.class)
61  			return (double)(Long)arg;
62  		if (arg.getClass() == String.class) {
63  			try {
64  				return Double.parseDouble((String)arg);
65  			} catch (final Exception e) {
66  				throw new IllegalArgumentException("can't convert \"" + arg + "\" to a floating point number!");
67  			}
68  		}
69  		if (arg.getClass() == Boolean.class)
70  			return (Boolean)arg ? 1.0 : 0.0;
71  
72  		throw new IllegalArgumentException("can't convert argument to a floating point number!");
73  	}
74  	/***
75  	 *  Assumes that "arg" is a "String", "Boolean", "Long" or a "Double and converts it to "double".
76  	 *  @return the converted argument as a "double"
77  	 *  @throws IllegalArgumentException if the argument cannot be converted to a "double"
78  	 */
79  	static public String getArgAsString(final Object arg) throws IllegalArgumentException {
80  		if (arg.getClass() == Boolean.class)
81  			return (Boolean)arg ? "TRUE" : "FALSE";
82  		if (arg.getClass() == Double.class) {
83  			final String asString = arg.toString();
84  			int length = asString.length();
85  
86  			// Strip trailing zeros:
87  			while (length > 1 && asString.charAt(length - 1) == '0')
88  				--length;
89  
90  			// Strip trailing period or comma:
91  			if (asString.charAt(length - 1) == '.' || asString.charAt(length - 1) == ',')
92  				--length;
93  
94  			return asString.substring(0, length);
95  		}
96  
97  		return arg.toString();
98  	}

125 	 *  @return the converted argument as a "boolean"
126 	 *  @throws IllegalArgumentException if the argument cannot be converted to a "boolean"

128 	static public boolean getArgAsBoolean(final Object arg) throws IllegalArgumentException {
129 		if (arg.getClass() == Double.class) {
130 			final double d = (Double)arg;
131 			return d == 0.0 ? false : true;
132 		}
133 		if (arg.getClass() == Long.class) {
134 			final long l = (Long)arg;
135 			return l == 0L ? false : true;
136 		}
137 		if (arg.getClass() == String.class) {
138 			final String argAsString = (String)arg;
139 
140 			if (argAsString.equalsIgnoreCase(Boolean.valueOf(true).toString()))
141 				return true;
142 			if (argAsString.equalsIgnoreCase(Boolean.valueOf(false).toString()))
143 				return false;
144 			throw new IllegalArgumentException("can't convert \"" + argAsString + "\" to a boolean!");
145 		}
146 		if (arg.getClass() == Boolean.class)
147 			return (Boolean)arg;
148 
149 		throw new IllegalArgumentException("can't convert argument to a boolean!");
150 	}

152 	/***
153 	 *  Carefully adds the numbers in "a" minimising loss of precision.
154 	 *
155 	 *  @return the sum of the elements of "a"
156 	 */
157 	static public double numericallySafeSum(final double a[]) {
158 		int positiveCount = 0;
159 		for (double d : a) {
160 			if (d >= 0.0)
161 				++positiveCount;
162 		}
163 
164 		// Separate positive and negative values:
165 		final double[] positiveValues = new double[positiveCount];
166 		final double[] negativeValues = new double[a.length - positiveCount];
167 		int positiveIndex = 0;
168 		int negativeIndex = 0;
169 		for (double d : a) {
170 			if (d >= 0.0)
171 				positiveValues[positiveIndex++] = d;
172 			else
173 				negativeValues[negativeIndex++] = d;
174 		}
175 
176 		double positiveSum = 0.0;
177 		if (positiveValues.length > 0) {
178 			// Add values in increasing order of magnitude:
179 			Arrays.sort(positiveValues);
180 			for (double d : positiveValues)
181 				positiveSum += d;
182 		}
183 
184 		double negativeSum = 0.0;
185 		if (negativeValues.length > 0) {
186 			// Add values in increasing order of magnitude:
187 			Arrays.sort(negativeValues);
188 			for (int i = negativeValues.length - 1; i >= 0; --i)
189 				negativeSum += negativeValues[i];
190 		}
191 
192 		return positiveSum + negativeSum;
193 	}

199 		if ((i % 100) == 11)
200 			return Integer.toString(i) + "th";
201 
202 		switch (i % 10) {
203 		case 1:
204 			return Integer.toString(i) + "st";
205 		case 2:
206 			return Integer.toString(i) + "nd";
207 		case 3:
208 			return Integer.toString(i) + "rd";
209 		default:
210 			return Integer.toString(i) + "th";
211 		}

214 	/***
215 	 *  @return the sample variance of the numbers in x[]
216 	 */
217 	static public double calcSampleVariance(final double[] x) {
218 		final int N = x.length;
219 		if (N < 2)
220 			throw new IllegalArgumentException("can't calculate a variance with fewer than 2 values!");
221 
222 		final double[] xSquared = new double[N];
223 		for (int i = 0; i < N; ++i)
224 			xSquared[i] = x[i] * x[i];
225 
226 		final double sumOfX = numericallySafeSum(x);
227 		final double sumOfXSquared = numericallySafeSum(xSquared);
228 
229 		return (sumOfXSquared - (sumOfX * sumOfX) / (double)N) / (double)(N - 1);
230 	}

296 	 *  Attempts to convert all arguments, including Lists to a uniform array of doubles.
297 	 *
298 	 *  @throws IllegalArgumentException if any scalar argument cannot be converted to a double or any list
299 	 *          argument contains an element that cannot be converted to a number.
300 	 */
301 	static public double[] getDoubles(final Object[] args) throws FunctionError {
302 		final List<Double> numbers = new ArrayList<Double>();
303 
304 		for (int i = 0; i < args.length; ++i) {
305 			final Object arg = args[i];
306 			if (arg instanceof List) {
307 				final List list = (List)arg;
308 				for (final Object listElement : list) {
309 					final Double d = convertToDouble(listElement);
310 					if (d == null)
311 						throw new FunctionError("can't convert list element \"" + listElement
312 						                        + "\" to a number!", i);
313 					numbers.add(d);
314 				}
315 			} else {
316 				final Double d = convertToDouble(arg);
317 				if (d == null)
318 					throw new FunctionError("can't convert \"" + arg + "\" to a number!", i);
319 				numbers.add(d);
320 			}
321 		}
322 
323 		final double[] doubles = new double[numbers.size()];
324 		int index = 0;
325 		for (final Double d : numbers)
326 			doubles[index++] = d;
327 
328 		return doubles;
329 	}

469 	 *  @return "arg" converted to a Boolean, if possible. else null
470 	 */
471 	static private Boolean convertToBoolean(final Object arg) {
472 		if (arg.getClass() == Double.class)
473 			return (Double)arg == 0.0 ? false : true;
474 		if (arg.getClass() == Long.class)
475 			return (double)(Long)arg == 0L ? false : true;
476 		if (arg.getClass() == Integer.class)
477 			return (double)(Integer)arg == 0 ? false : true;
478 		if (arg.getClass() == String.class) {
479 			final String s = (String)arg;
480 			if (s.equalsIgnoreCase("true"))
481 				return true;
482 			if (s.equalsIgnoreCase("false"))
483 				return false;
484 			return null;
485 		}
486 		if (arg.getClass() == Boolean.class)
487 			return (Boolean)arg;
488 
489 		return null;
490 	}

494 	 *  @return null if the translation failed, the input object if no translation was necessary or a new object if a successful translation was possible
495 	 */

497 		final Class type = input.getClass();
498 		if (type == Double.class || type == Long.class || type == Boolean.class || type == String.class)
499 			return input;

501 		if (type == Float.class)
502 			return new Double((Float)input);
503 
504 		if (type == Integer.class)
505 			return new Long((Integer)input);
506 
507 		if (type == Short.class)
508 			return new Long((Short)input);
509 
510 		if (type == Byte.class)
511 			return new Long((Byte)input);
512 
513 		if (type == Character.class)
514 			return new Long((Character)input);

517 		return null;
