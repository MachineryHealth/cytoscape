38  /***
39   *  A node in the parse tree.
40   */
41  public abstract class Node {
42  	private final int sourceLocation; // What location the "source code" is this associated with.
43  
44  	/*** Base class constructor for any <code>Node</code> type.
45  	 *  @param sourceLocation  start of the location in the equation where the code was found
46  	 *                         that was turned into a node in the parse tree
47  	 */
48  	public Node(final int sourceLocation) {
49  		this.sourceLocation = sourceLocation;
50  	}
51  
52  	/*** Returns the start of the location in the equation where the code was found that was
53  	 *  turned into a node in the parse tree.
54  	 *  @return the location in the source code that resulted in this node
55  	 */
56  	public final int getSourceLocation() {
57  		return sourceLocation;
58  	}
59  
60  	/*** Returns s <code>String</code> representation of this node.
61  	 *  @return a textual representation of this node
62  	 *  Note: This is typically only used for debugging.
63  	 */
64  	public abstract String toString();
65  
66  	/*** Returns the type of this node.
67  	 *  @return the type of the result of the code generated from this node will result in 
68  	 */
69  	public abstract Class getType(); // The type of this node.
70  
71  	/*** Returns the left child if it exists or null if it doesn't.
72  	 *  @return the left child, if any, of this node in the parse tree
73  	 */
74  	public abstract Node getLeftChild();
75  
76  	/*** Returns the right child if it exists or null if it doesn't.
77  	 *  @return the right child, if any, of this node in the parse tree
78  	 */
79  	public abstract Node getRightChild();
80  
81  	/*** Generated code for this node and pushes it onto the execution stack. */
82  	public abstract void genCode(final Stack<CodeAndSourceLocation> codeStack);
83  }
