33  public final class EquationUtil {
34  	private EquationUtil() { } // Exists to prevent creating instances of this class!
35  
36  	/***
37  	 *  @return "attribName" written as am attribute reference with a leading $-sign
38  	 */
39  	public static String attribNameAsReference(final String attribName) {
40  		if (isSimpleAttribName(attribName))
41  			return "$" + attribName;
42  		else
43  			return "${" + escapeAttribName(attribName) + "}";
44  	}
45  
46  	/***
47  	 *  @return "d" converted to a Long using Excelâ„¢ rules
48  	 *  @throws IllegalArgumentException if "d" is outside the range of a long
49  	 */
50  	public static long doubleToLong(final double d) {
51  		if (d > Long.MAX_VALUE || d < Long.MIN_VALUE)
52  			throw new IllegalArgumentException("floating point value is too large to be converted to a Long!");
53  
54  		double x = ((Double)d).longValue();
55  		if (x != d && d < 0.0)
56  			--x;
57  
58  		return (long)x;
59  	}
60  
61  	/***
62  	 *  @param attribName the name to test
63  	 *  @return true if "attribName" start with a letter and consists of only letters and digits, else false
64  	 */
65  	private static boolean isSimpleAttribName(final String attribName) {
66  		final int length = attribName.length();
67  		if (length == 0)
68  			throw new IllegalStateException("empty attribute names should never happen!");
69  
70  		if (!Character.isLetter(attribName.charAt(0)))
71  			return false;
72  
73  		for (int i = 1; i < length; ++i) {
74  			final char ch = attribName.charAt(i);
75  			if (!Character.isLetter(ch) && !Character.isDigit(ch))
76  				return false;
77  		}
78  
79  		return true;
80  	}
81  
82  	/***
83  	 *  @return "attribName" with characters that need to be backslash-escaped when written as
84  	 *           part of an attribute refernce, escaped
85  	 */
86  	private static String escapeAttribName(final String attribName) {
87  		final int length = attribName.length();
88  		final StringBuilder escapedAttribName = new StringBuilder(length * 2);
89  		for (int i = 0; i < length; ++i) {
90  			final char ch = attribName.charAt(i);
91  			switch (ch) {
92  			case ' ':
93  			case '//':
94  			case '{':
95  			case '}':
96  			case ':':
97  			case ',':
98  			case '(':
99  			case ')':
100 				escapedAttribName.append('//');
101 			}
102 			escapedAttribName.append(ch);
103 		}
104 
105 		return escapedAttribName.toString();
106 	}
