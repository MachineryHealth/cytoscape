33  import java.util.List;
34  import java.util.ArrayList;

37  /***
38   *  A class describing a function argument.
39   */
40  public final class ArgDescriptor {
41  	private ArgType argType;
42  	private String argName;
43  	private String description;
44  
45  	public ArgDescriptor(final ArgType argType, final String argName, final String description) {
46  		this.argType = argType;
47  		this.argName = argName;
48  		this.description = description;
49  	}
50  
51  	public ArgType getArgType() { return argType; }
52  	public String getArgName() { return argName; }
53  	public String getDescription() { return description; }
54  	public boolean isOptional() { return argType.isOptional(); }
55  
56  	public boolean isCompatibleWith(final Class type) {
57  		if (FunctionUtil.isSomeKindOfList(type))
58  			return isCompatibleList(type);
59  
60  		for (final Class compatibleType : argType.getCompatibleTypes()) {
61  			if (type == compatibleType)
62  				return true;
63  		}
64  
65  		return false;
66  	}
67  
68  	/***
69  	 *  @return true if "type" is a List type that specifies the element type otherwise we return false
70  	 */
71  	private boolean isSpecificList(final Class type) {
72  		return type == DoubleList.class || type == LongList.class || type == StringList.class || type == BooleanList.class;
73  	}
74  
75  	/***
76  	 *  @return true if "listType", which must be some type of List is a type compatible with this argument descriptor
77  	 */
78  	private boolean isCompatibleList(final Class listType) {
79  		// First we handle the case where "listType" is highly specific...
80  		if (isSpecificList(listType)) {
81  			for (final Class compatibleType : argType.getCompatibleTypes()) {
82  				if (compatibleType == listType || compatibleType == List.class)
83  					return true;
84  			}
85  			return false;
86  		}
87  
88  		// ...and here we handle the case where we don't have any information about the argument types of "listType":
89  		for (final Class compatibleType : argType.getCompatibleTypes()) {
90                          if (FunctionUtil.isSomeKindOfList(compatibleType))
91                                  return true;
92  		}
93  
94                  return false;
95  	}
96  
97  	/***
98  	 *  @return the types that are compatible with this argument
99  	 */
100 	public Class[] getCompatibleTypes() {
101 		return argType.getCompatibleTypes();
102 	}
103 
104 	public boolean acceptsMultipleArgs() {
105 		return argType.acceptsMultipleArgs();
106 	}
107 }
