\section{対数の定義}


\begin{displaymath}
y=a^{x}
\end{displaymath}

\noindent
の逆関数を

\begin{displaymath}
x=\log_{a}y
\end{displaymath}

\noindent
と表現する。この場合、\(x\)は\(a\)を底とする\(y\)の対数であるという。また
対数\(x\)において、\(y\)を真数という。10を底とした対数を常用対数、ネイピア数\(e\)を
底とした対数を自然対数という。

\section{対数の基本的な性質}

\subsection{定義から直ちに導かれる関係}

\begin{displaymath}
a^{\log_{a}x}=x
\end{displaymath}

\subsection{対数の和}

\begin{displaymath}
\log_{a}(xy)=\log_{a}x+\log_{a}y
\end{displaymath}


\subsection{対数の差}

\begin{displaymath}
\log_{a}\frac{x}{y}=\log_{a}x-\log_{a}y
\end{displaymath}

\subsection{真数の指数}

\begin{displaymath}
\log_{a}x^{p}=p\log_{a}x
\end{displaymath}

\subsection{底の変換}

\begin{displaymath}
\log_{a}x=\frac{\log_{b}x}{\log_{b}a}
\end{displaymath}
