\section{微分}

\subsection{微分の定義}

関数\(f(x)\)に対して

\begin{displaymath}
f'(x)=\frac{df(x)}{dx}=\lim_{\Delta x\to 0}\frac{f(x+\Delta x)-f(x)}{\Delta x}
\end{displaymath}

\noindent
で定義される\(f'(x)\)を\(f(x)\)の{\bf 導関数}という。また\(f(x)\)から
\(f'(x)\)を求めることを、\(f(x)\)を{\bf 微分}するという。図\ref{deriv1}に
示すように微分の幾何学的な意味は\(f(x)\)に対する接線の傾きである。

\begin{figure}
\begin{center}
\includegraphics[scale=0.5]{Figures/deriv1}
\end{center}
\vspace{-2em}
\caption{微分の幾何学的意味}
\label{deriv1}
\begin{quotation}
\begin{small}
\(f(x)\)上の2点の座標\((x,f(x))\)と\((x+\Delta x, f(x + \Delta x))\)を考えると
この2点を通る直線の傾きは、\(\frac{f(x+\Delta x)-f(x)}{\Delta x}\)である。\(f(x)\)
の微分とは、\(\Delta x\)が限りなく0に近づくときのこの直線の傾きの極限である。それは
\(f(x)\)における接線の傾きである。
\end{small}
\end{quotation}
\end{figure}

\subsection{変数のべき乗の微分}

$n$を任意の実数として、

\begin{displaymath}
(x^{n})^{\prime}=nx^{n-1}
\end{displaymath}

\subsection{関数の積・商の微分}

\begin{displaymath}
\{f(x)g(x)\}^{\prime}=f^{\prime}(x)g(x)+f(x)g^{\prime}(x)
\end{displaymath}

\begin{displaymath}
\left\{\frac{f(x)}{g(x)}\right\}^{\prime}=\frac{f^{\prime}(x)g(x)-f(x)g^{\prime}(x)}{g(x)^{2}}
\end{displaymath}

\subsection{合成関数の微分}

$y=f(u), u=g(x)$として、

\begin{displaymath}
\frac{dy}{dx}=\frac{dy}{du}\cdot\frac{du}{dx}
\end{displaymath}

\subsection{逆関数の微分}

\begin{displaymath}
\frac{dy}{dx}=\frac{1}{\frac{dx}{dy}}
\end{displaymath}

\subsection{媒介変数を持つ関数の商の微分}

$x=f(t), y=g(t)$のとき、

\begin{displaymath}
\frac{dy}{dx}=\frac{\frac{dy}{dt}}{\frac{dx}{dt}}=\frac{g^{\prime}(t)}{f^{\prime}(t)}
\end{displaymath}
