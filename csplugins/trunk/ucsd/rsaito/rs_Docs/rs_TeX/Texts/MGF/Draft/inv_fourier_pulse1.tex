長さ$ 2\tau$、大きさ$\displaystyle \frac{1}{2\tau}$のパルス\(f(t)\)のフーリエ変換は、

\begin{equation}
F(w)=\frac{1}{\tau w}\sin\tau w
\end{equation}

\noindent
となる。逆フーリエ変換は、

\begin{eqnarray}
f(t) & = & \frac{1}{2\pi}\int_{-\infty}^{\infty}\frac{1}{\tau w}\sin(\tau w)e^{iwt}dw \nonumber\\
& = & \frac{1}{2\pi}\int_{-\infty}^{\infty}\frac{1}{\tau w}\sin(\tau w)\left\{\cos wt+i\sin wt\right\}dw\nonumber\\
& = & \frac{1}{2\pi}\int_{-\infty}^{\infty}\frac{1}{\tau w}\sin\tau w\cos wtdw+\frac{i}{2\pi}\int_{-\infty}^{\infty}\frac{1}{\tau w}\sin\tau w\sin wtdw\nonumber\\
& = & \frac{1}{2\pi}\int_{-\infty}^{\infty}\frac{1}{\tau w}\sin\tau w\cos wtdw\label{eq_pulse_inv_fourier10}
% & = & \frac{1}{\pi}\int_{0}^{\infty}\frac{1}{\tau w}\sin\tau w\cos wtdw\nonumber\\
% & = & \frac{1}{2\pi}\int_{0}^{\infty}\frac{1}{\tau w}\sin(\tau+t)wdw
%      + \frac{1}{2\pi}\int_{0}^{\infty}\frac{1}{\tau w}\cos(\tau-t)wdw\nonumber
\end{eqnarray}

% & = & \frac{1}{\tau}\left[\log w\sin\tau w\cos wt\right]_{-\infty}^{\infty}-\frac{1}{\tau}\int_{-\infty}^{\infty}\frac{1}{w}(\sin\tau w\cos wt)^{\prime}dw\nonumber\\
% & = & \frac{1}{\tau}\left[\log w\sin\tau w\cos wt\right]_{-\infty}^{\infty}-\int\frac{1}{w}(\cos\tau w\cos wt-\sin\tau w\sin wt)dw\nonumber\\
% & = & \frac{1}{\tau}\left[\log w\sin\tau w\cos wt\right]_{-\infty}^{\infty}\nonumber
